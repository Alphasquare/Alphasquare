class Commands:
    commands = {}
    c_prefix = ""
    
    def __init__(self, prefix):
        self.c_prefix = prefix
    
    def register(self, command, handler):
        self.commands[command] = handler
    
    def is_registered(self, name):
        for command in self.commands:
            if command[0] == name:
                return True
        
        return False
    
    def has_enough_args(self, name, args):
        for command in self.commands:
            if command[0] == name:
                if command[1] <= args or command[1] == -1:
                    return True
                else:
                    return False
        
        return None
    
    def execute(self, name, args):
        for n in range(len(self.commands)):
            if self.commands.keys()[n-1][0] == name:
                self.commands[self.commands.keys()[n-1]](args)
